📝 What is Text Processing?

Text processing refers to the Analysis, Manipulation, and Generation of text. In C, it means working with character arrays and standard library functions. C has no built-in string type.

Key Tasks of Text Processing
TaskDescriptionExample

String Definition in C

A string in C is an array of characters terminated with '\0' (NULL). ASCII value of NULL = 0. Each character = 1 byte.

Key RuleWhen the compiler sees text in double quotes, it automatically appends '\0' at the end. Without '\0', C cannot know where the string ends.
CString definition examples 
char ch = 'x';          // character constant — single quotes
char ch[] = "x";        // string constant — double quotes, 2 bytes ('x' + '\0')

char a1[10];             // 10 bytes, garbage values — sizeof(a1) = 10
char a2[] = "atma";     // 5 bytes: 'a','t','m','a','\0'
char a3[10] = "atma";   // 10 bytes, remaining 6 filled with '\0'
char a4[] = "at\0ma";   // 6 bytes — print only "at" (stops at first '\0')
char a5[] = "atma\\0";  // 7 bytes — "atma\0" printed as is
⚙️ Initialization Rules
  • Size not specified: compiler counts elements and allocates exactly those bytes.
  • Size specified: unused locations filled with '\0' (partial initialization).
  • Hard-coded char arrays: programmer must manually add '\0' at end.
CAll initialization variants 
/* sizeof shown for 32-bit GCC */
char a2[] = {'a','t','m','a'};        // sizeof = 4, NO '\0' → unsafe to print
char a3[] = "atma";               // sizeof = 5, safe to print
char a4[10] = {'a','t','m','a'};  // sizeof = 10, partial init → safe
char a5[10] = "atma";             // sizeof = 10, safe
char a7[] = {'a','t','m','a','\0','t','r','i','s','h','a','\0'};
// sizeof = 12, prints only "atma" (stops at first '\0')
📥 Reading & Displaying Strings

Use %s format specifier with scanf/printf.

Importantscanf with %s stops at whitespace. Use %[^\n]s to read entire line including spaces.
Cscanf/printf variants 
char str3[100];
scanf("%s", str3);          // reads until whitespace → only first word

char str4[100], str5[100];
scanf("%s %s", str4, str5);  // reads two words separated by space

char str6[100];
scanf("%[^\n]s", str6);       // reads entire line (stops at newline)

char str7[100];
scanf("%[abcd]s", str7);      // only chars a,b,c,d allowed; stops at others
scanf("%[^abcd]s", str7);     // stops when a,b,c,d encountered

/* Loop-based printing */
int i = 0;
while(str3[i] != '\0') { printf("%c", str3[i]); i++; }

🛠️ User-Defined String Functions
CAll 6 versions of strlen 
/* V1: Array indexing */
int my_strlen(char str[]) {
    int i = 0;
    while(str[i] != '\0') { ++i; }
    return i;
}

/* V2: Pointer increment */
int my_strlen(char *str) {
    int i = 0;
    while(*str) { i++; str++; }
    return i;
}

/* V3: Recursion + pre-increment */
int my_strlen(char *str) {
    if(!(*str)) return 0;
    else return 1 + my_strlen(++str);
}

/* V4: Recursion, no pointer change */
int my_strlen(char *str) {
    if(!(*str)) return 0;
    else return 1 + my_strlen(str + 1);
}

/* V5: Recursion + post-increment (careful: infinite loop risk if used wrongly) */
int my_strlen(char *str) {
    if(!(*str)) return 0;
    else return 1 + my_strlen(str++);
}

/* V6: Pointer arithmetic */
int my_strlen(char *str) {
    char *copy = str;
    while(*copy) { copy++; }
    return copy - str;   // pointer subtraction = length
}
CUser-defined string functions 
/* my_strcpy — V4 (most concise) */
void my_strcpy(char *b, char *a) {
    while(*b++ = *a++);   // copies char by char including '\0'
}

/* my_strcmp */
int my_strcmp(char *a, char *b) {
    for(;*b && *a && *b == *a; a++, b++);
    return *a - *b;  // 0 = equal, <0 = a0 = a>b
}

/* my_strchr — find first occurrence */
char* my_strchr(char *a, char c) {
    while(*a && *a != c) { a++; }
    return !(*a) ? NULL : a;
}

/* my_strcat */
void my_strcat(char *a, char *b) {
    while(*a) { a++; }           // go to end of a
    while(*b) { *a = *b; a++; b++; }  // copy b
    *a = '\0';
}
⚠️ Common String Pitfalls & Best Practices
ErrorExampleFix
Buffer Overflowchar dest[5]; strcpy(dest,"Hello World!");Make buffer large enough; use strncpy
Missing '\0'char d[5]; strncpy(d,"Hello",5);Manually: d[sizeof(d)-1]='\0';

Uninitialized pointerchar *str; strcpy(str,"Hi");char *str = malloc(20);
Best Practices Summary1. Always include space for '\0'. 2. Prefer strncpy/strncat/snprintf. 3. Manually add '\0' if needed. 4. Check buffer size before copy. 5. Initialize pointers before use. 6. Use free() after malloc. 7. Validate strings before strlen. 8. Use memmove for overlapping buffers.
💻 Command Line Arguments (CLA)

Values passed to a program from the terminal when it runs. Allow dynamic, flexible input without hardcoding.

Signatureint main(int argc, char *argv[]) — argc = argument count (≥1, includes program name). argv[0] = program name. argv[1]…argv[argc-1] = actual args. All args are strings → convert with atoi(), atof().
CCLA Examples 
#include <stdio.h>
#include <stdlib.h>

/* Example 1: print all args */
int main(int argc, char *argv[]) {
    for(int i = 0; i < argc; i++)
        printf("Arg %d: %s\n", i, argv[i]);
    return 0;
}
/* Run: ./a.exe sindhu r pai  → 4 args */

/* Example 2: sum of integer args */
int main(int argc, char *argv[]) {
    int sum = 0;
    for(int i = 1; i < argc; i++)
        sum += atoi(argv[i]);   // converts string → int
    printf("Sum = %d\n", sum);
    return 0;
}
/* Run: ./a.exe 3 4 9 → Sum = 16 */

➕ malloc()

Prototype: void *malloc(size_t size);

Allocates the specified number of bytes on heap. Does NOT initialize — contains garbage values. Returns NULL on failure.

Cmalloc examples 
#include <stdio.h>
#include <stdlib.h>

/* Single integer */
int *p = (int*)malloc(sizeof(int));
*p = 10;
printf("p = %d\n", *p);  // 10

/* Array of n integers */
int n = 5;
int *x = (int*)malloc(n * sizeof(int));
if(x == NULL) { printf("Allocation failed!\n"); return 1; }
for(int i = 0; i < n; i++) scanf("%d", &x[i]);
for(int i = 0; i < n; i++) printf("%d\t", *(x+i));
free(x);
0️⃣ calloc()

Prototype: void *calloc(size_t nmemb, size_t size);

Allocates memory for nmemb elements of size bytes each and initializes all to zero. Returns NULL on failure.

Cmalloc vs calloc 
int *a = (int*)malloc(5 * sizeof(int));  // garbage values
int *b = (int*)calloc(5, sizeof(int));  // all zeros: 0 0 0 0 0
🔄 realloc()

Prototype: void *realloc(void *ptr, size_t size);

Resizes a previously allocated block. Preserves existing data (up to new size). Returns NULL on failure (original block untouched).

Special CasesIf ptr is NULL → behaves like malloc(size). If size = 0 → behaves like free(ptr). If realloc moves the block, old pointer becomes dangling!
Crealloc demo 
int *p = (int*)malloc(3 * sizeof(int));
/* ... fill p[0]=33, p[1]=22, p[2]=15 ... */

int new_size = 10;
int *temp = (int*)realloc(p, new_size * sizeof(int));
if(temp == NULL) {
    printf("realloc failed — original still valid\n");
} else {
    p = temp;  // safe to reassign only after NULL check
    /* new size > old: extra slots have garbage */
    /* new size < old: data beyond new size is lost */
}
🗑️ free()

Deallocates heap memory. The OS uses bookkeeping info stored at allocation time to know how many bytes to release. After free(), the pointer becomes a dangling pointer — set it to NULL.

Cfree usage 
int *x = (int*)malloc(sizeof(int));
*x = 10;
free(x);
x = NULL;  // best practice — avoid dangling pointer
// free(NULL) is safe — does nothing
🐛 Common DMM Errors
ErrorDescriptionFix
Dangling PointerPointer to freed/moved memorySet pointer = NULL after free
NULL PointerDereferencing NULL → guaranteed crashAlways check malloc return before use
Memory LeakAllocated memory with no pointer → garbageAlways free before reassigning pointer
Double Freefree() on already-freed memory → undefinedSet to NULL after free; free(NULL) is safe
Memory OverrunWrite beyond allocated boundsCheck index bounds always
Invalid Freefree() on stack/non-malloc'd memoryOnly free heap-allocated memory
Double Free WarningDO NOT attempt double free. It may corrupt the memory manager, cause crashes, or alter execution flow in unpredictable ways.
✅ DMM Best Practices
  • Always check returned pointer ≠ NULL before using.
  • Use calloc for zero-initialized arrays instead of malloc + manual init.
  • Free all allocated memory when no longer needed.
  • Set pointer to NULL after free to avoid dangling ptr.
  • Store realloc result in temp pointer, check NULL, then assign.
  • Use n * sizeof(type) — never just n.
  • Every malloc/calloc/realloc must have exactly one matching free.

📦 Why Structures?

A structure is a user-defined data type that groups variables of different types under a single name. Without structures, you'd need separate arrays for each field (error-prone implicit relationships).

CStructure syntax & examples 
/* Declaration (blueprint only — no memory allocated here) */
struct student {
    int  roll_no;
    char name[20];
    int  marks;
};

/* Variable declaration — memory allocated now */
struct student s1;       // stack allocation
struct student *s2;      // pointer — only pointer memory allocated

/* Assignment */
s1.roll_no = 101;
s1.marks   = 87;
strcpy(s1.name, "Rahul");

/* Initialization at declaration */
struct student s2 = {44, "abc", 100};
struct student s3 = {44, "abc"};          // partial → marks = 0
struct student s4 = {.name="abc", .roll_no=44}; // designated (C99+)
🎯 Accessing Members — Dot & Arrow Operators
OperatorUsed withSyntax
. (dot)Structure variables1.roll_no
-> (arrow)Pointer to structures2->roll_no  (same as (*s2).roll_no)
Member-wise CopyStructures of the same type are assignment-compatible: s2 = s1 copies each member individually. They do NOT point to the same memory.
💾 Memory Allocation & Padding

Processor architecture mandates alignment. On a 32-bit GCC machine, the word size is 4 bytes. The compiler may add padding to align members.

CPadding examples (32-bit GCC) 
struct test  { int i; char j; };          // 8 bytes (4+4 with padding)
struct test1 { char j; int i; };          // 8 bytes (4+4 with padding)
struct test2 { char k; char j; int i; }; // 8 bytes (4+4)
struct test3 { int i; char k; int j; }; // 12 bytes (4+4+4)

/* #pragma pack — control padding */
#pragma pack(1)   // no padding — tightly packed
struct PackedStruct { char a; int b; };   // 5 bytes (not 8)
#pragma pack()    // restore default

⚙️ Structures & Functions
Pass by ValueIn C, structs are always passed by value (member-wise copy). To modify a struct inside a function, pass a pointer to it.
CPass by value vs pointer 
struct Player { int id; char name[20]; };

/* read — needs pointer to modify original */
void read(struct Player *p1) {
    scanf("%d", &(p1->id));
    scanf("%[^\n]s", p1->name);
}

/* disp — pointer to const: read-only, efficient (no copy) */
void disp(const struct Player *p1) {
    printf("%d %s\n", p1->id, p1->name);
}

/* return struct from function */
struct Player modify(struct Player p) {
    strcpy(p.name, "Sourav");
    return p;  // returned as temp copy, assigned to p1 in caller
}
🗃️ Array of Structures

A collection of structure variables in contiguous memory. Perfect for managing multiple records.

CArray of structures — full example 
#include <stdio.h>
#include <string.h>

struct Student { int roll_no; char name[50]; float marks; };

void read(struct Student *s, int n) {
    for(int i = 0; i < n; i++) {
        scanf("%d", &s[i].roll_no);
        scanf(" %[^\n]", s[i].name);
        scanf("%f", &s[i].marks);
    }
}

void display(struct Student *s, int n) {
    for(int i = 0; i < n; i++)
        printf("%d  %s  %.2f\n", s[i].roll_no, s[i].name, s[i].marks);
}

int main() {
    struct Student s[100];
    int n; scanf("%d", &n);
    read(s, n);   // array degenerates to pointer
    display(s, n);
    return 0;
}

/* Pointer to array of structs */
struct Student *ptr = s;
// ptr++ moves by sizeof(struct Student) bytes
printf("%d", (ptr+2)->roll_no);  // third student's roll no
🧩 Bit Fields

Allow specifying the exact number of bits (not bytes) for structure members. Essential for embedded systems and hardware registers.

CBit field examples 
struct Flags {
    unsigned int isVisible : 1;  // 1 bit → 0 or 1
    unsigned int isEnabled : 1;  // 1 bit → 0 or 1
    unsigned int errorCode : 3;  // 3 bits → 0–7
};  // total 5 bits; max 1 byte vs 12 bytes without bit fields!

struct Status {
    int bin1 : 4;               // signed 4-bit: -8 to +7
    unsigned int bin2 : 2;      // unsigned 2-bit: 0–3
};

/* Unnamed bit field for padding */
struct example {
    unsigned int a : 3;
    unsigned int   : 2;  // 2 unnamed padding bits
    unsigned int b : 4;
};

/* Zero-width forces alignment to next word boundary */
struct s { int x:4; int :0; unsigned int y:2; };
// size = 8 bytes (not 4) because :0 forces new word

/* Invalid: cannot take address of bit field, no arrays, no pointers */
// int a[5] : 3;   → ERROR
// int *ptr : 4;   → ERROR

🤝 Unions

A union is a user-defined type where all members share the same memory location. At any time, only one member holds a valid value. Size = size of largest member.

CUnion fundamentals 
union car {
    char name[10];   // 10 bytes
    float price;     // 4 bytes
};
// sizeof(union car) = 10 (largest member)

union car c;
strcpy(c.name, "Benz");
printf("%s\n", c.name);  // Benz ✅
c.price = 4000000.00;
printf("%f\n", c.price);  // 4000000.00 ✅
printf("%s\n", c.name);   // CORRUPTED ❌ (price overwrote name)

/* All fields have offset 0 */
printf("%lu\n", offsetof(union A, y));  // 0 for union
printf("%lu\n", offsetof(struct B, y)); // 4 for struct
⚖️ Union vs Structure
Featurestructunion
Memory per variableSum of all members (+ padding)Size of largest member
Member storageEach member has unique locationAll members share same location
Address of membersAscending (different offsets)All same (offset 0)
Access at one timeAll members simultaneouslyOnly one member valid at a time
Modify one, affect others?NoYes
Use CaseUnions save memory when data members are mutually exclusive — e.g., a student can be identified by name OR roll number but not both needed at once.
🔢 Enumerations (enum)

Enum provides a set of named integer constants to replace magic numbers. Starts at 0 by default, increments by 1. Values are not stored in memory — they're compile-time symbols.

CEnum examples 
enum months { jan, feb, mar, apr, may, jun, jul, aug, sep, oct, nov, dec };
// jan=0, feb=1, ... dec=11

enum day { sunday=1, monday, tuesday=5, wednesday, thursday=10, friday, saturday };
// 1, 2, 5, 6, 10, 11, 12

enum State { Working=1, Failed=0, Freezed=0 };
// Multiple names can share same value

/* Use in switch to print symbol name */
void printing(enum example1 e1) {
    switch(e1) {
        case abc: printf("abc"); break;
        case bef: printf("bef"); break;
        default:  printf("not found");
    }
}

/* Design flags — values outside range allowed in C (not Java) */
enum design_flags { bold=1, italics=4, underline=8 };
int df = italics | bold;  // 5 — not defined in enum, but valid in C
if(df & bold)    printf("bold\n");
if(df & italics) printf("italics\n");
Enum RulesConstants must be unique in scope. Cannot take address of enum constants. Cannot modify constant values (mar++ = error). Enum variables can be changed. No string printing — use switch. Enum constants must be integer constants (not 3.5 or "hello").

🔁 Self-Referential Structures

A structure that contains a pointer to itself as a data member. The pointer's size is fixed at compile time (4 or 8 bytes), so the compiler can determine the struct's total size.

CSelf-referential struct demos 
/* Cannot have variable of same struct type inside */
// struct A { int a; struct A a1; };  → COMPILE ERROR

/* CAN have pointer to same type */
struct Sample { int a; struct Sample *p; };

struct Sample s;
s.a = 100;
s.p = &s;               // points to itself
printf("%d %d %d\n", s.a, s.p->a, s.p->p->a);
// 100 100 100 — all same node

/* Node template for linked list */
struct Node { int data; struct Node *next; };
🔗 Singly Linked List

A chain of nodes in non-contiguous memory, logically connected by pointers. Sequential access only — no random access. Insertion/deletion at any position requires no element shifting.

TypeDescription
Singly Linked ListEach node → next node. Last node → NULL.
Doubly Linked ListEach node → next & previous. Two pointers per node.
Circular Linked ListLast node → first node (forms a cycle).
CSingly linked list — create, display, free 
#include <stdio.h>
#include <stdlib.h>

struct node { int info; struct node *link; };
typedef struct node NODE_T;

/* Display all nodes */
void display(NODE_T *q) {
    while(q != NULL) {
        printf("%d\t", q->info);
        q = q->link;
    }
}

/* Free all nodes */
void freelist(NODE_T *q) {
    NODE_T *r;
    while(q != NULL) {
        printf("\n%d deleted", q->info);
        r = q->link;  // save next before freeing current
        free(q);
        q = r;
    }
}

int main() {
    NODE_T *s = (NODE_T*)malloc(sizeof(NODE_T));
    s->info = 100;
    s->link = (NODE_T*)malloc(sizeof(NODE_T));
    s->link->info = 200;
    s->link->link = (NODE_T*)malloc(sizeof(NODE_T));
    s->link->link->info = 300;
    s->link->link->link = NULL;

    display(s);   // 100  200  300
    freelist(s);  // 100 deleted  200 deleted  300 deleted
    s = NULL;     // avoid dangling pointer
    return 0;
}
⚡ Priority Queue

A queue where each element has an associated priority. Higher-priority elements are served first (regardless of order of insertion).

OperationQueuePriority Queue
Insert (enqueue)Add to rearAdd with priority
Remove (dequeue)Remove from frontRemove highest-priority element

Types of Priority Queue
  • Ascending Priority Queue: Smallest number = highest priority.
  • Descending Priority Queue: Largest number = highest priority.

CPriority Queue using ordered linked list 
#include <stdio.h>
#include <stdlib.h>

typedef struct node {
    int data;
    int priority;
    struct node *next;
} PQ_Node;

PQ_Node *head = NULL;

/* Enqueue: insert by descending priority */
void enqueue(int data, int priority) {
    PQ_Node *newNode = (PQ_Node*)malloc(sizeof(PQ_Node));
    newNode->data = data;
    newNode->priority = priority;
    newNode->next = NULL;

    if(head == NULL || head->priority < priority) {
        newNode->next = head;
        head = newNode;
    } else {
        PQ_Node *curr = head;
        while(curr->next && curr->next->priority >= priority)
            curr = curr->next;
        newNode->next = curr->next;
        curr->next = newNode;
    }
}

/* Dequeue: remove highest priority (front) */
int dequeue() {
    if(head == NULL) { printf("Queue empty!\n"); return -1; }
    PQ_Node *temp = head;
    int val = temp->data;
    head = head->next;
    free(temp);
    return val;
}

int main() {
    enqueue(10, 2);
    enqueue(20, 5);  // highest priority
    enqueue(30, 1);
    printf("%d\n", dequeue());  // 20 (priority 5)
    printf("%d\n", dequeue());  // 10 (priority 2)
    printf("%d\n", dequeue());  // 30 (priority 1)
    return 0;
}
