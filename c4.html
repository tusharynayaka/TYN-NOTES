Lecture 1

Introduction to File Handling

Data created during program execution is stored temporarily in variables. Once the program ends, this data is lost. To preserve data after termination, files are used.

File handling allows a C program to create, read, write, and modify files stored on disk — enabling persistent storage, data sharing between programs, and efficient handling of large datasets.

📁 Why File Handling?
  • Persistent data storage — survives beyond program life
  • Data sharing between programs
  • Handle large datasets that may not fit in RAM

Real-Life Examples

  • Saving user input / game scores
  • Creating system logs & audit trails
  • Reading configuration files on startup
  • Writing scientific simulation results
  • Recording financial transactions securely

File Classification

TypeExtensionStorageHuman-Readable?Security
Text File.txtASCII characters✅ YesLess secure (plain text)
Binary File.binBinary (0s & 1s)❌ NoMore secure, less space
Key Concepts
  • Physical Name — the actual filename on disk managed by the OS
  • Logical Name (FileHandle) — the C identifier (pointer) used in program
  • Mode — read, write, append, or combinations

Opening & Closing Files — fopen / fclose

fopen()

Opens a file and returns a FILE* pointer. Returns NULL on failure.

ModeDescription
"r"Read only — file must exist
"w"Write only — creates new / overwrites existing
"a"Append — creates if not exists, adds to end
"r+"Read + Write — file must exist
"w+"Read + Write — creates new / overwrites
"a+"Read + Append — creates if not exists

fclose()

Closes the file stream, flushes buffers, and frees associated memory. Returns 0 on success, EOF on error.

⚠️ Best Practice
Always check if fp == NULL after fopen(). Initializing a FILE pointer does not load the file into memory.
Example 1 — Demo of fopen and fclose 
#include<stdio.h>
int main()
{
    FILE* fp = fopen("data.txt", "r");  // file must exist to open in "r" mode
    if (fp == NULL)
        printf("File cannot be opened\n");
    else {
        printf("File can be accessed for reading\n");
        fclose(fp);   // always close after use
    }
    return 0;
}
Sample Output
File can be accessed for reading

{

        printf("File cannot be opened\n");

#include<stdio.h>
int main() {
    FILE *fp1 = fopen("data.txt", "r");
    FILE *fp2 = fopen("out.txt",  "w");
    char ch[500];
    if (fp1 == NULL || fp2 == NULL)
        printf("Issue in opening the file\n");
    else {

        fclose(fp2);
    }
    return 0;
}

Used primarily for binary files — reads/writes entire structs or arrays in one call.

Example 6 — fwrite & fread: Employee Records 
#include<stdio.h>
#include<stdlib.h>
struct employee {
    char  name[15];
    char  designation[15];
    int   age;
    float salary;
} emp;

int main() {
    FILE *fp = fopen("employee_data.bin", "wb");
    if (fp == NULL) { printf("Error\n"); return 1; }
    char ch;
    do {
        printf("\nEnter name: ");       scanf("%s", emp.name);
        printf("Enter designation: "); scanf("%s", emp.designation);
        printf("Enter age: ");         scanf("%d", &emp.age);
        printf("Enter salary: ");      scanf("%f", &emp.salary);
        fwrite(&emp, sizeof(emp), 1, fp);
        printf("Add another? (y/n): ");
        fflush(stdin);  ch = getchar();
    } while (ch == 'y' || ch == 'Y');
    fclose(fp);

    fp = fopen("employee_data.bin", "rb");
    printf("\n\tNAME\tDESIG\tAGE\tSALARY\n");
    while (fread(&emp, sizeof(emp), 1, fp) > 0)
        printf("\t%s\t%s\t%d\t%.2f\n",
               emp.name, emp.designation, emp.age, emp.salary);
    fclose(fp);
    return 0;
}

Random Access — fseek / ftell / rewind

Jump to any specific position in a file without sequential reading.

FunctionPrototypeDescription

ftelllong ftell(FILE *fp)Returns current file pointer position
rewindvoid rewind(FILE *fp)Moves pointer to beginning of file

whence values for fseek:

ConstantValueMeaning
SEEK_SET0From beginning of file
SEEK_CUR1From current position
SEEK_END2From end of file
Example 8 — fseek and ftell 
#include<stdio.h>
int main() {
    FILE *fp = fopen("data.txt", "r");
    if (fp == NULL) { printf("Cannot open\n"); return 1; }

    printf("Initially pointer is at %ld\n", ftell(fp));

    fseek(fp, 5, SEEK_SET);   // move 5 bytes from beginning
    printf("After SEEK_SET(5): %ld\n", ftell(fp));

    fseek(fp, 2, SEEK_CUR);   // move 2 bytes from current
    printf("After SEEK_CUR(2): %ld\n", ftell(fp));

    fseek(fp, -9, SEEK_END);  // 9 bytes before end
    printf("After SEEK_END(-9): %ld\n", ftell(fp));
    printf("Character at position: %c\n", fgetc(fp));

    fclose(fp);
    return 0;
}
Sample Output

Lecture 2

Error Handling in File Handling

⚠️ Common File Errors
  1. Reading beyond EOF — no more data can be read
  2. Operating on unopened file — file must be opened first
  3. Writing to a read-only file — mode mismatch
  4. Invalid filename — bad file name or type
  5. Write-protected file — OS permission denied

C does not have built-in exception handling. It uses manual error handling via return values, global errno, and helper functions.

1. Return Value Checking

Most C file functions return a value indicating success or failure — always check it!

Return Value Checking Cases 
/* Case 1: fopen() returns NULL on failure */
FILE *fp = fopen("data.txt", "r");
if (fp == NULL) {
    printf("Error opening file\n");
    exit(1);
}

/* Case 2: fscanf() returns items read; check expected count */
int num;
if (fscanf(fp, "%d", &num) != 1)
    fprintf(stderr, "Error reading integer\n");

/* Case 3: fclose() returns 0 on success */
if (fclose(fp) != 0)
    perror("Error closing file");

2. Global Variable errno

When a C function fails, it sets the global variable errno (defined in <errno.h>) to a code identifying the error type.

errno ValueError Type
0No Error
1Operation not permitted
2No such file or directory
5I/O error
9Bad file number
12Out of memory
13Permission denied

4. feof() / ferror() / clearerr()

FunctionPrototypeReturnsPurpose

💡 Best Practice
Check the return value of fgetc() / fscanf() first. Use feof() only to confirm EOF after a failed read, not before reading. Use ferror() alongside perror() for detailed error info.
Example 4 — feof() usage 
#include<stdio.h>
int main() {
    FILE *fp = fopen("data.txt", "r");
    char ch;
    if (fp != NULL) {
        while (!feof(fp)) {
            ch = fgetc(fp);
            if (!feof(fp))   // check again after reading
                putchar(ch);
        }
    }
    fclose(fp);
    return 0;
}
Example 5 — ferror() detects illegal read on write-only file 
#include<stdio.h>
int main() {

        if (fgets(buffer, sizeof(buffer), fp) != NULL)
            printf("Read: %s", buffer);
        else if (ferror(fp))
            printf("An error occurred while reading the file.\n");

Lecture 4

Searching File Data using Array of Pointers

An array of pointers (AOP) is a powerful technique for working with file data in memory. Instead of working with the data directly, we store addresses of each record — enabling fast search and manipulation without altering original data.

Why Array of Pointers?
  • Flexible memory management
  • Faster search — manipulate pointers, not data
  • Easier data manipulation without altering original file
  • Efficient in-memory operations (sort, search, filter)

Binary Search on File Integers

Example 1 — Recursive Binary Search with AOP on integers from file 
#include<stdio.h>
#include<stdlib.h>

int Search(int **aop, int low, int high, int key);

int main() {
    FILE *fp = fopen("numbers.txt", "r");
    if (fp == NULL) { printf("Can't open\n"); return 1; }

    int a, n = 0;
    while (fscanf(fp, "%d", &a) != -1)  n++;

    int  *p   = malloc(n * sizeof(int));
    int **aop = (int **)malloc(n * sizeof(int *));
    for (int i = 0; i < n; i++)  aop[i] = &p[i];

    rewind(fp);
    int i = 0;
    while (fscanf(fp, "%d", &p[i]) != -1)  i++;
    fclose(fp);

    printf("Enter integer to search: ");
    int key;
    scanf("%d", &key);

    int res = Search(aop, 0, n - 1, key);
    if (res == -1)  printf("Not found\n");
    else           printf("Found at index %d\n", res);
    return 0;
}

int Search(int **aop, int low, int high, int key) {
    if (low > high) return -1;
    int mid = (low + high) / 2;
    if      (*aop[mid] == key) return mid;
    else if (key < *aop[mid]) return Search(aop, low, mid - 1, key);
    else                       return Search(aop, mid + 1, high, key);
}

Binary Search on Array of Structs (CSV)

Example 2 — Search student by roll number (sorted_data.csv) 
#include<stdio.h>
#include<string.h>
#include<stdlib.h>

typedef struct student { int roll; char name[20]; } STUDENT;

void init_ptr(STUDENT s[], STUDENT *p[], int n) {
    for (int i = 0; i < n; i++)  p[i] = &s[i];
}
void disp(STUDENT *p[], int n) {
    for (int i = 0; i < n; i++)
        printf("%d %s\n", p[i]->roll, p[i]->name);
}
int Search(STUDENT **aop, int low, int high, int key) {
    if (low > high) return -1;
    int mid = (low + high) / 2;
    if      (aop[mid]->roll == key) return mid;
    else if (key < aop[mid]->roll) return Search(aop, low, mid-1, key);
    else                            return Search(aop, mid+1, high, key);
}

int main() {
    FILE *fr = fopen("sorted_data.csv", "r");
    if (!fr) { printf("Can't open\n"); exit(1); }
    char a[200]; fgets(a, 200, fr); // skip header
    STUDENT s[100]; int i = 0;
    while (fgets(a, 200, fr)) {
        char *item = strtok(a, ",");
        s[i].roll  = atoi(item);
        item       = strtok(NULL, ",");
        strcpy(s[i].name, item);
        i++;
    }
    int n = i; fclose(fr);

    STUDENT *p[100];
    init_ptr(s, p, n);
    disp(p, n);

    int key;
    printf("Enter roll number to search: ");
    scanf("%d", &key);
    int res = Search(p, 0, n-1, key);
    if (res == -1)
        printf("Student not found\n");
    else
        printf("Roll: %d, Name: %s", p[res]->roll, p[res]->name);
    return 0;
}
Lecture 5

Sorting File Data using Array of Pointers

When sorting file content with an AOP, we swap pointers, not the actual data. This reduces memory operations and improves performance significantly on large datasets.

Key Idea
Read data → store in dynamic array → point AOP elements to each item → sort AOP (swap pointers only) → write sorted AOP to new file.

Sort Integers from File

Example 1 — Selection Sort integers using AOP, write to sorted file 
#include<stdio.h>
#include<stdlib.h>

void swap(int **lhs, int **rhs) {
    int *temp = *lhs;  *lhs = *rhs;  *rhs = temp;
}
void sort(int **aop, int n) {
    for (int i = 0; i < n-1; i++) {
        int pos = i;
        for (int j = i+1; j < n; j++)
            if (*aop[pos] > *aop[j])  pos = j;
        if (pos != i)  swap(&aop[pos], &aop[i]);
    }
}

int main() {
    FILE *fp = fopen("numbers.txt", "r");
    int a, n = 0;
    while (fscanf(fp, "%d", &a) != -1)  n++;

    int  *p   = malloc(n * sizeof(int));
    int **aop = (int **)malloc(n * sizeof(int *));
    for (int i = 0; i < n; i++)  aop[i] = &p[i];

    rewind(fp);
    int i = 0;
    while (fscanf(fp, "%d", &p[i]) != -1)  i++;

    sort(aop, n);

    FILE *fpw = fopen("sorted_data.txt", "w");
    for (int i = 0; i < n; i++)
        fprintf(fpw, "%d\t", *aop[i]);
    printf("Writing done.\n");
    fclose(fp); fclose(fpw);
    return 0;
}

Sort Structs — by roll / by name

Example 2 — Menu-driven sort (roll_no / name) on student CSV 
#include<stdio.h>
#include<string.h>
#include<stdlib.h>
struct student { int roll; char name[20]; };

void swap(struct student **lhs, struct student **rhs) {
    struct student *t = *lhs;  *lhs = *rhs;  *rhs = t;
}
void selection_sort_roll_no(struct student *s[], int n) {
    for (int i = 0; i < n-1; i++) {
        int pos = i;
        for (int j = i+1; j < n; j++)
            if (s[pos]->roll > s[j]->roll)  pos = j;
        if (pos != i)  swap(&s[pos], &s[i]);
    }
}
void selection_sort_names(struct student *s[], int n) {
    for (int i = 0; i < n-1; i++) {
        int pos = i;
        for (int j = i+1; j < n; j++)
            if (strcmp(s[pos]->name, s[j]->name) > 0)  pos = j;
        if (pos != i)  swap(&s[pos], &s[i]);
    }
}

Lecture 7

Header Files in C

Header files enable modular programming by separating interface (declarations) from implementation (definitions).

Advantages
  • Avoid rewriting code across multiple .c files
  • Separate declarations from definitions — clean & organized
  • Easier maintenance & collaboration on large projects

Execution Steps with Header Files:

Compilation & Linking Commands
# Step 1: Compile each .c file into an object file
gcc -c palin.c palin_client.c      # produces palin.o and palin_client.o

# Step 2: Link all .o files into executable
gcc palin.o palin_client.o         # produces a.out or a.exe

# Step 3: Run
./a.out

Palindrome Example — Client / Server Split

palin.h — Interface (Header File) 
// palin.h — contains only the declaration (interface)
int is_palin(int);
palin.c — Server (Implementation) 
// palin.c — implementation of is_palin
int is_palin(int n) {
    int rev = 0, num = n;
    while (n > 0) {
        rev = rev * 10 + n % 10;
        n  /= 10;
    }
    return num == rev;
}
palin_client.c — Client Code 
#include<stdio.h>
#include"palin.h"   // user-defined header — double quotes
int main() {
    int n;
    printf("Enter number: ");
    scanf("%d", &n);
    if (is_palin(n))
        printf("%d is a palindrome\n", n);
    else
        printf("%d is not a palindrome\n", n);
    return 0;
}
Sample Outputs
Enter number: 121 → 121 is a palindrome Enter number: 123 → 123 is not a palindrome

employee_server.c — Implementation 
#include"employee.h"
#include<stdio.h>

void scan_details(employee_t *e, int n) {
    for (int i = 0; i < n; i++)
        scanf("%s %d %d", (e+i)->name, &(e+i)->id, &(e+i)->basic_salary);
}
void display_details(const employee_t *e, int n) {
    for (int i = 0; i < n; i++)
        printf("%s\t%d\t%d\n", (e+i)->name, (e+i)->id, (e+i)->basic_salary);
}
employee_t max(employee_t *e, int n) {
    int mx = 0; employee_t emax;
    for (int i = 0; i < n; i++)
        if ((e+i)->basic_salary > mx) {
            mx = (e+i)->basic_salary;  emax = *(e+i);
        }
    return emax;
}

Env Vars

Environment Variables

Environment variables are dynamic named values that affect running processes. Programs inherit them from the shell.

Context of program execution can be known via two ways: command-line arguments and environment variables.

Extended main signature
int main(int argc, char *argv[], char *envp[])

#include<stdio.h>
int main(int argc, char *argv[], char *envp[])
{
    for (int i = 0; envp[i] != NULL; i++)

💡 Important Notes
  • Names are case-sensitive and cannot contain =
  • Values can be any string; no embedded null characters

  • Setting PATH directly can break execution — concatenate to existing PATH
Preprocessor

Preprocessor Directives

C compilation is a multi-stage process: Preprocessing → Compilation → Loading → Linking → Execution. The C Preprocessor (CPP) is a text substitution tool that runs before compilation.

#DirectivePurpose
1#defineSubstitutes a preprocessor macro
2#includeInserts a header file
3#undefUndefines a macro
4#ifdefTrue if macro is defined
5#ifndefTrue if macro is NOT defined
6#if / #else / #elif / #endifConditional compilation
7#errorPrints error to stderr
8#pragmaCompiler-specific special commands
Tip
Use gcc -E file.c to see the effect of the preprocessor — it shows you exactly what code gets compiled after all substitutions.

Macros with #define

A macro is a named piece of code. When CPP encounters the name, it performs plain text substitution — no type, no memory allocation.

⚠️ Macro vs Variable
No memory is allocated for a macro. &MAX is an error. Macros cannot be changed with =.
Programs 1–7 — Macro examples 
#include<stdio.h>
#define PI        3.14              // symbolic constant
#define MAX       10               // no memory, &MAX is ERROR
#define STR       "Hello All"     // string macro
#define SUM(a,b)  a+b             // macro with parameter
#define sqr(x)   (x)*(x)         // always wrap args in () !
#define cube(x)  sqr(x)*(x)      // macro inside macro

int main() {
    printf("PI = %f\n", PI);
    printf("STR = %s\n", STR);
    printf("SUM(3,4) = %d\n", SUM(3,4));
    printf("sqr(5) = %d\n", sqr(5));
    printf("sqr(2+3) = %d\n", sqr(2+3));  // (2+3)*(2+3) = 25
    printf("cube(3) = %d\n", cube(3));
    return 0;
}
Output
PI = 3.140000 STR = Hello All SUM(3,4) = 7 sqr(5) = 25 sqr(2+3) = 25 cube(3) = 27

    printf("Date: %s\n", __DATE__);
    printf("Time: %s\n", __TIME__);

Enum vs Macros
  • Macro has no type; enum constants have type int
  • Macro substitution happens at preprocessing; enum at compilation
  • Macro can be redefined (warning); enum constants cannot be redefined

Conditional Compilation

Compile only specific blocks of code based on conditions — especially useful for platform-specific code.

Programs 1–6 — #ifdef, #ifndef, #if, #elif 
#include<stdio.h>
#define MAX 20

int main() {
    /* #ifdef: included only if MAX is defined */
#ifdef MAX
    printf("MAX defined\n");
#endif

    /* #ifndef: included only if MAX is NOT defined */
#ifndef MIN
    printf("MIN not defined\n");
#endif

    /* #if with expression */
#if (2 > 1)
    printf("2 > 1 is true\n");
#endif

    /* #if + #else */
#if ((2+3) < 2)
    printf("Hello\n");
#else
    printf("Hi\n");
#endif

    /* #elif chain */
#if (12 % 5 == 0)
    printf("Multiple of 5\n");
#elif (12 % 3 == 0)
    printf("Multiple of 3\n");
#endif

    return 0;
}
Output
MAX defined MIN not defined 2 > 1 is true Hi Multiple of 3

Pragma Directives

Compiler-specific directives — ignored if not supported (no error).

PragmaPurpose
#pragma startup / exitRun functions before/after main()
#pragma warn -rvlSuppress "missing return value" warning

Pragma 1 — #pragma startup and exit 
#include<stdio.h>
void func1(); void func2();
#pragma startup func1
#pragma exit func2
void func1() { printf("Inside func1()\n"); }
void func2() { printf("Inside func2()\n"); }
int main()  { printf("Inside main()\n");  return 0; }
Output
Inside func1() Inside main() Inside func2()
Pragma 3 — #pragma pack(1): struct size = 9 bytes 
#include<stdio.h>
#pragma pack(1)    // align to 1 byte → no padding
struct sample {
    int  a;   // 4 bytes
    char b;   // 1 byte (normally +3 padding without pragma)
    int  x;   // 4 bytes
};
int main() {
    printf("%lu\n", sizeof(struct sample));  // 9 (not 12)
    return 0;
}
Output
9

Portable Program Development

Using conditional compilation and pre-defined platform macros, code can adapt to different operating systems at compile time.

SystemMacroValue
Windows (MinGW)__MINGW32__1
Linux/Unix__unix__1
macOS__APPLE__1
Portable Program — Platform-specific fflush / fpurge 
#include<stdio.h>
int main() {
#ifdef __MINGW32__
    printf("Windows system\n");
    char c, d;
    scanf("%c", &c);  fflush(stdin);
    scanf("%c", &d);
    printf("%c %c\n", c, d);
#elif __unix__
    printf("Unix/Linux system\n");
    char c, d;
    scanf("%c", &c);  // use fpurge(stdin) on Linux
    scanf("%c", &d);
    printf("%c %c\n", c, d);
#elif __APPLE__
    printf("macOS system\n");
#else
    printf("Other system\n");
#endif
    return 0;
}
